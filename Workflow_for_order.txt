@startuml
actor Actor
collections Backend
database Redis
database PostgreSQL
database MongoDB
database Neo4j
title : Workflow for Selecting and Confirming an Order

activate Actor
loop Until user clicks "confirm order"
Actor -> Backend : Looks at a specific item


activate Backend
Backend -> Neo4j : Request recommended items
activate Neo4j
Neo4j -> Actor : Response with recommended \nitems
deactivate Neo4j

Actor -> Backend : User clicks "add to cart"

Backend -> Redis : The given item is added to cart
activate Redis

Redis -> Backend : Sends items from \ncart to backend
end
deactivate Redis
deactivate Backend

Actor -> Backend : User clicks on \n"view shopping cart"
activate Backend

Backend -> PostgreSQL : Requests information on product \nnumber(s) from PostgreSQL

activate PostgreSQL
PostgreSQL -> Backend : Response with all information \nabout given item(s)
deactivate PostgreSQL

Backend -> Actor : User sees confirm orderpage
deactivate Backend
Actor -> Backend: User clicks "confirm order"
activate Backend

Backend -> Redis : Fetches users session data from cart
activate Redis
Redis -> Backend : Returns users session data

group Two-Phased Commit
Backend -> PostgreSQL : Prepares transaction to <b>update \n<b>quantity</b> on affected item(s)
activate PostgreSQL
PostgreSQL -> Backend : Success

Backend -> MongoDB : Inserts order
activate MongoDB
MongoDB -> Backend : Success
deactivate MongoDB

Backend -> PostgreSQL : Commits prepared transaction

deactivate PostgreSQL
end

Backend -> Redis : Deletes cart when used
deactivate Redis

Backend -> Neo4j : Creates order and relations to items
activate Neo4j
Neo4j -> Backend : Success
deactivate Neo4j

Backend -> Actor : Sends order confirmation \nto user

deactivate Backend
deactivate Actor
@enduml
